En komponent för att segmentera information och minska den mängd som direkt presenteras för användaren.

```tsx
import { Tabs, TabList, Tab, TabPanel } from '@midas-ds/components'
```

```tsx
<Tabs>
  <TabList aria-label='Viktig information om frukter och bär'>
    <Tab id='vitaminer'>...</Tab>
    <Tab id='frukter'>...</Tab>
    <Tab id='hallon'>...</Tab>
  </TabList>
  <TabPanel id='vitaminer'>...</TabPanel>
  <TabPanel id='frukter'>...</TabPanel>
  <TabPanel id='hallon'>...</TabPanel>
</Tabs>
```

## Beskrivning

Midas `Tabs` är en wrapperkomponent som används tillsammans med `TabList`, `Tab` och `TabPanel`.
Samtliga komponenter utgår från React Arias komponenter med samma namn, vänligen se [dokumentationen för Tabs hos React Aria](https://react-spectrum.adobe.com/react-aria/Tabs.html) för en djupare förklaring.

## Varianter

Tabs finns i varianterna `"uncontained" | "contained"`, där den senare passar bra för äldre appar som använder tabs mer som "flikar":

### Contained

```tsx {1}
<Tabs variant='contained'>
  <TabList aria-label='Viktig information om frukter och bär'>
    <Tab id='vitaminer'>...</Tab>
    <Tab id='frukter'>...</Tab>
    <Tab id='hallon'>...</Tab>
  </TabList>
  <TabPanel id='vitaminer'>...</TabPanel>
  <TabPanel id='frukter'>...</TabPanel>
  <TabPanel id='hallon'>...</TabPanel>
</Tabs>
```

## Användning

### Standardval (okontrollerad)

Välj vilket alternativ som ska vara förvalt med attributet `defaultSelectedKey`, använd id för den flik som ska vara förvald.

```tsx {1}
<Tabs defaultSelectedKey='frukter'>
  <TabList aria-label='Viktig information om frukter och bär'>
    <Tab id='vitaminer'>...</Tab>
    <Tab id='frukter'>...</Tab>
    <Tab id='hallon'>...</Tab>
  </TabList>
  <TabPanel id='vitaminer'>...</TabPanel>
  <TabPanel id='frukter'>...</TabPanel>
  <TabPanel id='hallon'>...</TabPanel>
</Tabs>
```

### Kontrollerat läge

Valet av flik kan kontrolleras med hjälp av attributet `selectedKey` tillsammans med eventet `onSelectionChange`.
Flikens ID skickas tillbaka i callbacken vid valändring, vilket kan användas för att uppdatera ditt state.

```tsx
import React from 'react'
import type { Key } from 'react-aria-components'
```

```tsx
const [selectedTab, setSelectedTab] = React.useState<Key>('frukter')

return (
  <Tabs
    // highlight-start
    selectedKey={selectedTab}
    onSelectionChange={setSelectedTab}
    // highlight-end
  >
    <TabList aria-label='Viktig information om frukter och bär'>
      <Tab id='vitaminer'>...</Tab>
      <Tab id='frukter'>...</Tab>
      <Tab id='hallon'>...</Tab>
    </TabList>
    <TabPanel id='vitaminer'>...</TabPanel>
    <TabPanel id='frukter'>...</TabPanel>
    <TabPanel id='hallon'>...</TabPanel>
  </Tabs>
)
```

### Flikar med stängningsknapp

Här är ett exempel på flikar där varje flik har en stängningsknapp. Detta är användbart för scenarion där flikar kan läggas till eller tas bort dynamiskt av användaren. För mer information om dynamiska flikar, se [React Aria dynamic tabs](https://react-spectrum.adobe.com/react-aria/Tabs.html#dynamic-items).

```tsx
import React, { useState } from 'react'
import { X } from 'lucide-react'
import { Tab, TabList, Tabs, TabPanel, Text, Button } from '@midas-ds/components'

export const ClosableTabsExample = () => {
  const [openTabs, setOpenTabs] = useState([
    { id: 'bananer', title: 'Bananer' },
    { id: 'applen', title: 'Äpplen' },
  ])

  const handleCloseTab = (idToClose: string) => {
    if (openTabs.length > 1) setOpenTabs(prevTabs => prevTabs.filter(tab => tab.id !== idToClose))
  }

  return (
    <Tabs>
      <TabList>
        {openTabs.map(tab => (
          <Tab
            key={tab.id}
            id={tab.id}
            style={{ display: 'flex', gap: '0.5rem', alignItems: 'center' }}
          >
            {tab.title}
            <Button
              variant='icon'
              onPress={() => handleCloseTab(tab.id)}
            >
              <X size={16} />
            </Button>
          </Tab>
        ))}
      </TabList>
      {openTabs.map(tab => (
        <TabPanel
          key={tab.id}
          id={tab.id}
        >
          <Text>Information om {tab.title}</Text>
        </TabPanel>
      ))}
    </Tabs>
  )
}
```
