```tsx
import { List, ListItem } from '@midas-ds/components'
```

`List` bygger på [React Aria GridList](https://react-spectrum.adobe.com/react-aria/GridList.html) och ärver dess props, bland annat `disallowEmptySelection`, `renderEmptyState` och `disabledBehavior`.

## Varianter

### Utan val

Listar rader utan urvalsbeteende.

```tsx
<List aria-label="Sökresultat" label="Sökresultat">
  <ListItem id="1">Isabella W</ListItem>
  <ListItem id="2">Margareta B</ListItem>
  <ListItem id="3">Jenny F</ListItem>
</List>
```

### Enkelval

Sätt `selectionMode="single"` för enkelval med radioindikator.

```tsx
<List
  aria-label="Välj person"
  label="Välj person"
  selectionMode="single"
  onSelectionChange={keys => console.log(keys)}
>
  {items.map(item => (
    <ListItem key={item.id} id={item.id}>{item.name}</ListItem>
  ))}
</List>
```

### Flerval

Sätt `selectionMode="multiple"` för flerval med kryssrutor; `showSelectAll` lägger till en markera-alla-kryssruta i headern.

```tsx
import React from 'react'
import type { Selection } from 'react-aria-components'

const MyList = () => {
  const [selected, setSelected] = React.useState<Selection>(new Set())

  return (
    <List
      aria-label="Välj personer"
      label="Välj personer"
      selectionMode="multiple"
      showSelectAll
      headerSlot={({ selectedCount, totalCount }) => (
        <span>{selectedCount} av {totalCount} valda</span>
      )}
      selectedKeys={selected}
      onSelectionChange={setSelected}
    >
      {items.map(item => (
        <ListItem key={item.id} id={item.id}>{item.name}</ListItem>
      ))}
    </List>
  )
}
```

### Zebraränder

Alternerande radfärg via `striped`.

```tsx
import { List, ListItem, Tag, TagGroup, TagList } from '@midas-ds/components'

<List
  aria-label="Sökresultat"
  label="Sökresultat"
  headerSlot={<span>{items.length} poster</span>}
  striped
>
  {items.map(item => (
    <ListItem key={item.id} id={item.id} textValue={item.name}>
      {item.name}
      <ListItem.Trailing>
        <TagGroup aria-label="Status">
          <TagList>
            <Tag id="status" color={item.status === 'Aktiv' ? 'green' : 'red'}>
              {item.status}
            </Tag>
          </TagList>
        </TagGroup>
      </ListItem.Trailing>
    </ListItem>
  ))}
</List>
```

### Trailing-innehåll

Placera innehåll som ska visas till höger i `ListItem.Trailing` — det hanterar marginal och mellanrum mellan flera element automatiskt.

```tsx
import { Button, List, ListItem, Tag, TagGroup, TagList } from '@midas-ds/components'
import { Star } from 'lucide-react'

<List aria-label="Sökresultat" label="Sökresultat">
  {items.map(item => (
    <ListItem key={item.id} id={item.id} textValue={item.name}>
      {item.name}
      <ListItem.Trailing>
        <TagGroup aria-label="Status">
          <TagList>
            <Tag id="status">{item.status}</Tag>
          </TagList>
        </TagGroup>
        <Button variant="icon" aria-label={`Åtgärd för ${item.name}`} onPress={() => {}}>
          <Star size={16} />
        </Button>
      </ListItem.Trailing>
    </ListItem>
  ))}
</List>
```

### `onAction`

`onAction` aktiveras på enkelt klick utan `selectionMode`; med `selectionMode="single"` aktiveras det på dubbelklick.

```tsx
<List
  aria-label="Sökresultat"
  label="Sökresultat"
  onAction={key => console.log(key)}
>
  {items.map(item => (
    <ListItem key={item.id} id={item.id}>{item.name}</ListItem>
  ))}
</List>
```

<div className='card'>
  <RowActionExample />
</div>

:::info Beteende med urval

`onAction` och `selectionMode` kan kombineras, men beteendet skiljer sig:

- **Utan `selectionMode`** — `onAction` aktiveras på enkelt klick.
- **`selectionMode="multiple"`** — klick på raden aktiverar `onAction`; klick på kryssrutan väljer raden.
- **`selectionMode="single"`** — enkelt klick väljer raden; `onAction` aktiveras på dubbelklick.

:::

### Radåtgärder — enkel knapp

Enkelt klick väljer raden, dubbelklick aktiverar `onAction`, och knappen i raden är helt fristående från radmarkeringen.

```tsx
import { Button, List, ListItem } from '@midas-ds/components'
import { Star } from 'lucide-react'
import type { Selection } from 'react-aria-components'

const [selected, setSelected] = useState<Selection>(new Set())
const [rowAction, setRowAction] = useState<string | null>(null)
const [buttonAction, setButtonAction] = useState<string | null>(null)

<List
  aria-label="Sökresultat"
  label="Sökresultat"
  selectionMode="single"
  selectedKeys={selected}
  onSelectionChange={setSelected}
  onAction={key => setRowAction(String(key))}
  striped
>
  {items.map(item => (
    <ListItem key={item.id} id={item.id} textValue={item.name}>
      {item.name}
      <ListItem.Trailing>
        <Button variant="icon" aria-label={`Åtgärd för ${item.name}`} onPress={() => setButtonAction(item.name)}>
          <Star size={16} />
        </Button>
      </ListItem.Trailing>
    </ListItem>
  ))}
</List>
```

### Radåtgärder — kontextmeny

`MenuTrigger` och `Menu` i `ListItem.Trailing` för flera åtgärder per rad.

```tsx
import {
  Button, List, ListItem,
  Menu, MenuItem, MenuPopover, MenuTrigger,
  Tag, TagGroup, TagList,
} from '@midas-ds/components'
import { EllipsisVertical } from 'lucide-react'

<List
  aria-label="Sökresultat"
  label="Sökresultat"
  headerSlot={<span>{items.length} poster</span>}
  striped
>
  {items.map(item => (
    <ListItem key={item.id} id={item.id} textValue={item.name}>
      {item.name}
      <ListItem.Trailing>
        <TagGroup aria-label="Status">
          <TagList>
            <Tag id="status" color="sky">{item.status}</Tag>
          </TagList>
        </TagGroup>
        <MenuTrigger>
          <Button variant="icon" aria-label={`Åtgärder för ${item.name}`}>
            <EllipsisVertical size={20} />
          </Button>
          <MenuPopover>
            <Menu>
              <MenuItem id="edit">Redigera</MenuItem>
              <MenuItem id="delete">Ta bort</MenuItem>
            </Menu>
          </MenuPopover>
        </MenuTrigger>
      </ListItem.Trailing>
    </ListItem>
  ))}
</List>
```
